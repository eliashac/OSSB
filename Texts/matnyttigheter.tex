% Elias Hachichou

\poemtitle{Tredimensionella-tidsberoende Schrödingerekvationen}
\begin{equation}
i \hbar \frac{\partial}{\partial t} \Psi (\textbf{r},t) = - \frac{\hbar^2}{2m} \nabla^2 \Psi (\textbf{r},t) + V(\textbf{r}) \Psi (\textbf{r},t)
\end{equation}

\poemtitle{Endimensionella-tidsoberoende Schrödingerekvationen}
\begin{equation}
- \frac{\hbar^2}{2m} \frac{d^2}{d x^2} \psi (x) + V(x) \psi (x) = E \psi (x)
\end{equation}

\poemtitle{Newton-Raphson}
\begin{equation}
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)}
\end{equation}
