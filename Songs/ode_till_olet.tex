% O.D.E. till ölet

\settowidth{\versewidth}{Pi-pi-pi pilsnern som jag vet.}

\poemtitle{O.D.E. till ölet}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Mel. Trampa på gasen}\\!

\flagverse{1.}
Tu-tu-tu Tuborg\\*
och ca-ca-ca Carlsberg,\\*
det är den bästa\\*
Pi-pi-pi pilsnern som jag vet.\\!

%------------------------------------------------

\flagverse{2.}
Tu-tu-tu Carlsberg\\*
och ca-ca-ca Tuborg,\\*
det är den bästa\\*
pi-pi-pi ölet som jag vet.\\!

%------------------------------------------------

\flagverse{3.}
Tu-tu-tu Ölberg\\*
och ca-ca-ca Pilsborg,\\*
det är den bästa\\*
pi-pi-pi biran som jag vet.\\!

%------------------------------------------------

\flagverse{4.}
Tu-ca-pi ölsner\\*
och pi-ta-ca biradet\\*
det är den bästa\\*
ca-pi-tu lering som jag gjort\\!

%------------------------------------------------

\end{verse}
