\poemtitle{Liljekonvaljen}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{1.}
/:Liljekonvaljen,\\*
vackrast utav blommor\.\*
Liljekonvaljen som gömts i snön,\\*
när det blir tö kommer...:/\\!

%------------------------------------------------

\end{verse}
