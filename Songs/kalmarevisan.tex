% Kalmarevisan

\settowidth{\versewidth}{Hej dickom dickom dack!}

\poemtitle{Kalmarevisan}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Sångledare: S}\\*
\emph{Alla: A}\\!

\flagverse{S.}
Uti Kalmare stad\\*
ja där finns ingen kvast\\!

\flagverse{A.}
förrän lördagen!\\!

\flagverse{S.}
Hej dick...\\!

\flagverse{A.}
...hej dack!\\!

\flagverse{S.}
Jag slog i...\\!

\flagverse{A.}
...och vi drack\\!

\flagverse{S.}
Hej dickom dickom dack!\\!

\flagverse{A.}
Hej dickom dickom dack!\\!

\flagverse{A.}
För uti Kalmare stad,\\*
ja där finns det ingen kvast\\*
förrän lördagen!\\!

\flagverse{S.}
|:När som bonden kommer hem\\!

\flagverse{A.}
kommer bondekvinnan ut:|\\*
och är så stor i sin trut.\\!

%------------------------------------------------

\flagverse{S.}
Hej dick...\\!

\flagverse{A.}
...hej dack! etc.\\!

\flagverse{S.}
|:Var är pengarna du fått?\\!

\flagverse{A.}
Jo, dem har jag supit opp:|\\*
uppå Kalmare slott\\!

\flagverse{S.}
Hej dick...\\!

\flagverse{A.}
...hej dack! etc.\\!

\flagverse{S.}
|:Jag skall mäla dig an\\!

\flagverse{A.}
för vår kronbefallningsman:|\\*
och du skall få skam.\\!

\flagverse{S.}
Hej dick...\\!

\flagverse{A.}
...hej dack! etc.\\!

\flagverse{S.}
|:Kronbefallningsmannen vår\\!

\flagverse{A.}
satt på krogen i går:|\\*
och var full som ett får.\\!

\flagverse{S.}
Hej dick...\\!

\flagverse{A.}
...hej dack! etc.\\!

\end{verse}
