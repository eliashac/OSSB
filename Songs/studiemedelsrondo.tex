\poemtitle{Studiemedelsrondo}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Mel. Lossa sand}\\!

%------------------------------------------------

\flagverse{1.}
\emph{(Upprepas, allt snabbare)}\\*
Vi dricker punsch till lunch\\*
När vi har fått avin\\*
Vi lunchar hela dagen\\*
Tills kassan gått i sin\\!

%------------------------------------------------

\end{verse}
