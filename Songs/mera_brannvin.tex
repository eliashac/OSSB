% Mera brännvin

\settowidth{\versewidth}{mera marscher till Skåne,}

\poemtitle{Mera brännvin}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Mel: Internationalen}\\!

\flagverse{}
Mera brännvin i glasen,\\*
mera glas på vårt bord,\\*
mera bord på kalasen,\\*
mer kalas på vår jord.\\*
Mera jordar med måne,\\*
mera månar i mars,\\*
mera marscher till Skåne,\\*
mera Skåne, bevars, bevars, bevars!\\!

%------------------------------------------------

\poemauthorcenter{Premiärframförd å\\ Gastronomiska\\ Klubben i Uppsala 1963}

%------------------------------------------------

\end{verse}
