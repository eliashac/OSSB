\settowidth{\versewidth}{Slån i glaset gubbar,}

\poemtitle{Gammalt brännvin}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Mel. Gubben Noak}\\!

\flagverse{1.}
Gammalt brännvin,\\*
Gammalt brännvin\\*
finnes i vårt krus.\\*
Slån i glaset gubbar,\\*
många stora nubbar\\*
Gån till verket,\\*
gån till verket!\\*
Tag ett stadigt rus!\\!

%------------------------------------------------

\flagverse{2.}
Gammalt brännvin,\\*
gammalt brännvin\\*
flödar i vår kropp.\\*
Genom torra strupar\\*
rinna starka supar.\\*
Ge åt kroppen,\\*
ge åt själen\\*
kraft och frejdigt hopp!\\!

%------------------------------------------------

\end{verse}
