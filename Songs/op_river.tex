\settowidth{\versewidth}{Ja, OP river!}
\poemtitle{O.P River}

%------------------------------------------------

\begin{verse}[\versewidth]

\flagverse{}
\emph{Mel. Ol' Man River}\\!

\flagverse{1.}
OP river!\\*
Ja, OP river!\\*
Var gång jag lenat,\\*
in hals med renat.\\*
Jag sagt med iver,\\*
att OP river\\*
långt ner,\\*
långt ner.\\!

%------------------------------------------------

\poemauthorcenter{ Oscar Hammerstein II\\ Jerome Kern, 1927}

%------------------------------------------------

\end{verse}
